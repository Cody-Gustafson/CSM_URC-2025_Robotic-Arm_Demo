Install Git:
sudo apt update
sudo apt install git

Install ROS2 if Needed:
sudo apt update
sudo apt install curl gnupg lsb-release
sudo curl -sSL https://raw.githubusercontent.com/ros/rosdistro/master/ros.asc | sudo tee /usr/share/keyrings/ros-archive-keyring.gpg > /dev/null
echo "deb [signed-by=/usr/share/keyrings/ros-archive-keyring.gpg] http://packages.ros.org/ros2/ubuntu $(lsb_release -cs) main" | sudo tee /etc/apt/sources.list.d/ros2.list
sudo apt update
sudo apt install ros-humble-desktop

Install Moveit:
sudo apt install ros-humble-moveit

Navigate to Moveit Workspace:
cd ~/ws_moveit

Clone Repository:
git clone https://github.com/Cody-Gustafson/CSM_URC-2025_Robotic-Arm_Demo.git

Copy to Workspace:
cp -r CSM_URC-2025_Robotic-Arm/src/* ~/ws_moveit/src/

Copy to Arduino (If not Loaded):
cp -r CSM_URC-2025_Robotic-Arm/arduino/* <Arduino Folder>

Rebuild / Resource
cd ~/ws_moveit
colcon build
source install/setup.bash

Locate Arduino Port: (/dev/ttyUSB0 or /dev/ttyACM0)
ls /dev/tty*

Add Current User to Dialout:
sudo usermod -a -G dialout $USER
reboot

Install Serial Dependencies:
sudo apt update
sudo apt install python3-colcon-common-extensions
sudo apt install python3-serial

Install pyserial if Needed:
pip3 install pyserial
